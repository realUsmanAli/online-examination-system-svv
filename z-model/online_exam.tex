[STUDENT]

STATUS ::= LoggedOut | ExamActive | Submitted

OnlineExam
students : ℙ STUDENT
submitted : ℙ STUDENT
status : STUDENT \pfun STATUS

submitted ⊆ students
∀ s : STUDENT | s ∈ submitted ⇒ status(s) = Submitted
∀ s : STUDENT | s ∈ students \ submitted ⇒ status(s) = ExamActive
